#version 330 core
layout (triangles) in;
//一根顶点法线两个点，三根法线六个点
layout (line_strip, max_vertices = 6) out;	//几何着色器的输出是线

in VS_OUT {
    vec3 normal;
} gs_in[];

const float MAGNITUDE = 0.2;		//用来限制法向量大小

uniform mat4 projection;

void GenerateLine(int index){
    gl_Position = projection * gl_in[index].gl_Position;	//转换到裁剪空间
    EmitVertex();
    gl_Position = projection * (gl_in[index].gl_Position + vec4(gs_in[index].normal, 0.0) * MAGNITUDE);	//从顶点开始沿着法线加上一段距离
    EmitVertex();
    EndPrimitive();
}

void main(){
    GenerateLine(0);
    GenerateLine(1);
    GenerateLine(2);
}
